package pattern;

public class Factory {  //Фабрика
	public static void main(String[] args) {
	Factory factory = new Factory();           //Запуск фабрики
	Car toyota = factory.create("Toyota");
	Car audi = factory.create("Audi");
	toyota.drive();
	audi.drive();
	}
}

interface Car {          //Інтерфейс машина
	void drive();    //Метод
}

class Toyota implements Car {             //Машина: "Toyota"
	@Override 
	public void drive() {
		System.out.println("drive Toyota");
	}
}

class Audi implements Car {              //Машина: "Audi"
	@Override 
	public void drive() {
		System.out.println("drive Audi");
	}
}

class Factory {
	public Car create(String typeOfCar) {    //тип продукту: машина
		switch (typeOfCar) {
			case "Toyota" : return new Toyota();       //Якщо замовник замовить "Toyota"
			case "Audi": return new Audi();            //Якщо замовник замовить "Audi"
			default: return null;
		}
	}
}
